include(${CMAKE_SOURCE_DIR}/cmake/QT_helper.cmake)
# 自动生成 MOC 文件（Qt 元对象编译）
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)
# 设置QT工具源路径
set(QT_SRC_DIR "C:/Qt/6.10.1")
# 添加 Qt6 安装路径
if(CMAKE_CXX_COMPILER_ID STREQUAL "GNU") 
    # 适配 GCC (MinGW)
    message(STATUS "Detected GCC: Using standard mingw_64 path")
    set(CMAKE_PREFIX_PATH "C:/Qt/6.10.1/mingw_64")
elseif(CMAKE_CXX_COMPILER_ID MATCHES "Clang")
    # 适配 Clang (包括 Clang-cl 或 llvm-mingw)
    message(STATUS "Detected Clang: Using llvm-mingw_64 path")
    set(CMAKE_PREFIX_PATH "C:/Qt/6.10.1/llvm-mingw_64")
else()
    message(WARNING "Unknown compiler: ${CMAKE_CXX_COMPILER_ID}. Defaulting to mingw_64")
    set(CMAKE_PREFIX_PATH "C:/Qt/6.10.1/mingw_64")
endif()
find_package(Qt6 REQUIRED COMPONENTS Core Widgets)

add_executable(nn_webserver
    ${CMAKE_CURRENT_SOURCE_DIR}/nn_webserver/nn_webserver_api.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/nn_webserver/nn_webserver.cpp
)
add_json_support(nn_webserver
    LINKS
        json_net_libs
    FILE_DIR
        "${CMAKE_CURRENT_SOURCE_DIR}/nn_webserver"
)
add_nn_support(nn_webserver
    MAIN_DIR    
        ${CORE_DIR} # 为 src/core
    DEFINES
        USE_DEBUG 
        USE_THREAD
        DEBUG_ASSERT_H
    INCLUDES
        ${CMAKE_SOURCE_DIR}/api
    LINKS
        math_libs
)
add_web_support(nn_webserver)
add_json_support(nn_webserver
    FILE_DIR
        "${CMAKE_CURRENT_SOURCE_DIR}/nn_webserver"
)

add_executable(QT_api 
    ${CMAKE_CURRENT_SOURCE_DIR}/nn_QT/QT_nn_api.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/nn_QT/QT_nn_interface.cpp
)
add_QT_support(QT_api)
add_nn_support(QT_api
    MAIN_DIR    
        ${CORE_DIR} # 为 src/core
    DEFINES
        USE_DEBUG 
        USE_THREAD
        DEBUG_ASSERT_H
    INCLUDES
        ${CMAKE_SOURCE_DIR}/api
    LINKS
        math_libs
)
