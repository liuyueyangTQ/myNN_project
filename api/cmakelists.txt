include(${CMAKE_SOURCE_DIR}/cmake/QT_helper.cmake)
# 自动生成 MOC 文件（Qt 元对象编译）
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTORCC ON)
find_package(Qt6 REQUIRED COMPONENTS Core Widgets)

add_executable(nn_webserver
    ${CMAKE_CURRENT_SOURCE_DIR}/nn_webserver/nn_webserver_api.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/nn_webserver/nn_webserver.cpp
)
add_json_support(nn_webserver
    LINKS
        json_net_libs
    FILE_DIR
        "${CMAKE_CURRENT_SOURCE_DIR}/nn_webserver"
)
add_nn_support(nn_webserver
    MAIN_DIR    
        ${CORE_DIR} # 为 src/core
    DEFINES
        USE_DEBUG 
        USE_THREAD
        DEBUG_ASSERT_H
    INCLUDES
        ${CMAKE_SOURCE_DIR}/api
    LINKS
        math_libs
)
add_web_support(nn_webserver)
add_json_support(nn_webserver
    FILE_DIR
        "${CMAKE_CURRENT_SOURCE_DIR}/nn_webserver"
)

add_executable(QT_api 
    ${CMAKE_CURRENT_SOURCE_DIR}/nn_QT/QT_nn_api.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/nn_QT/QT_nn_interface.cpp
)
add_QT_support(QT_api)
add_nn_support(QT_api
    MAIN_DIR    
        ${CORE_DIR} # 为 src/core
    DEFINES
        USE_DEBUG 
        USE_THREAD
        DEBUG_ASSERT_H
    INCLUDES
        ${CMAKE_SOURCE_DIR}/api
    LINKS
        math_libs
)

add_subdirectory(nn_communication)
