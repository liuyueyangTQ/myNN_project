cmake_minimum_required(VERSION 3.10)
project(Metrix_test)

# 设置C++标准
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
# 1. 先设置构建类型为 Debug（关键！CMake 会自动适配 Debug 模式的默认编译选项）
# set(CMAKE_BUILD_TYPE Debug CACHE STRING "Build type" FORCE)
# 🔥 强制开启 Release 模式
set(CMAKE_BUILD_TYPE "Release" CACHE STRING "Build type" FORCE)
# 🔥 开启最高级优化
set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -O3 -march=native")

# 2. 强制为 C/C++ 设置 Debug 模式的编译选项（覆盖默认，确保 -g -O0 生效）
# 针对 C++ 文件
set(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} -g -O0")
# 如果项目包含 C 文件，补充这行
set(CMAKE_C_FLAGS_DEBUG "${CMAKE_C_FLAGS_DEBUG} -g -O0")

# 对于MinGW，需要设置正确的编译器标志
if(MINGW)
    set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -static-libgcc -static-libstdc++")
endif()
# 编译阶段：添加-flto !!不能加-flto，否则mingw与静态链接不兼容
# if(CMAKE_CXX_COMPILER_ID MATCHES "GNU|Clang")
#     set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -O3 -flto -march=native")
#     链接阶段：添加-flto
#     set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} -flto")
#     set(CMAKE_SHARED_LINKER_FLAGS "${CMAKE_SHARED_LINKER_FLAGS} -flto")
# endif()

# 引入封装的函数文件
include(${CMAKE_SOURCE_DIR}/cmake/basicTool_helper.cmake)
include(${CMAKE_SOURCE_DIR}/cmake/nn_helper.cmake)
include(${CMAKE_SOURCE_DIR}/cmake/protobuf_helper.cmake)
include(${CMAKE_SOURCE_DIR}/cmake/openGL_helper.cmake)
include(${CMAKE_SOURCE_DIR}/cmake/net_helper.cmake)

set(CORE_DIR "${CMAKE_SOURCE_DIR}/src/core")
set(TOOLS_DIR "${CMAKE_SOURCE_DIR}/src/tools")
message(STATUS "CORE_DIR: ${CORE_DIR}")
message(STATUS "TOOLS_DIR: ${TOOLS_DIR}")
# 设置静态库输出目录（关键！指定 build 下的文件夹）
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)

# 收集库源文件
file(GLOB MATH_LIB_SOURCES 
    ${CORE_DIR}/math/metrix.cpp
    ${CORE_DIR}/tensor/tensor.cpp
    ${CORE_DIR}/tensor/dtensor.cpp
    ${CORE_DIR}/tensor/ops.cpp
    ${CORE_DIR}/thread/threadpool.cpp
    ${CORE_DIR}/nn/nn.cpp
    ${CORE_DIR}/utils/buffer_util.cpp
)

# ==== 生成 math_libs 静态库 ====
add_library(math_libs STATIC ${MATH_LIB_SOURCES})
# 指定宏
target_compile_definitions(math_libs PUBLIC USE_DEBUG)
# 指定包含文件路径
target_include_directories(math_libs PUBLIC
    ${CORE_DIR}/common
    ${CORE_DIR}/math
    ${CORE_DIR}/tensor
    ${CORE_DIR}/thread
    ${CORE_DIR}/nn
    ${CORE_DIR}/utils
)

# ==== 生成 proto_libs 静态库 ====
add_library(proto_libs STATIC
    ${TOOLS_DIR}/protobuf/network.pb.cc
    ${TOOLS_DIR}/protobuf/proto_utils.cpp
)
add_protobuf_support(proto_libs
    DEFINES
        USE_DEBUG
    LINKS
        math_libs
)

# ==== 生成 json_net_libs 静态库 ====
add_library(json_net_libs STATIC
    ${TOOLS_DIR}/json/json_tool.cpp
)
add_json_support(json_net_libs
    DEFINES
        USE_DEBUG
)

# ==== 生成 netService_libs 静态库 ====
add_library(netService_libs STATIC
    ${TOOLS_DIR}/netService/server.cpp
)

add_subdirectory(src)
add_subdirectory(api)
